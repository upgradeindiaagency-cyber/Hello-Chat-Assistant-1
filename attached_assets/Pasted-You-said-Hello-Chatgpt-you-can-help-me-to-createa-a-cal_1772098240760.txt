🎛️ STEP 4: Controller Logic
📄 controllers/callController.js

const OpenAI = require("openai");
const fs = require("fs");
const Call = require("../models/Call");

const openai = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY
});

exports.uploadCall = async (req, res) => {
    try {
        const { salespersonName, clientName } = req.body;
        const audioPath = req.file.path;

        // 1️⃣ Transcribe Audio (Whisper)
        const transcription = await openai.audio.transcriptions.create({
            file: fs.createReadStream(audioPath),
            model: "whisper-1"
        });

        const transcriptText = transcription.text;

        // 2️⃣ Analyze Transcript using GPT
        const analysisPrompt = `
        Analyze this sales call transcript (Hindi + English + Gujarati mixed).

        Give response in JSON format:
        {
          greetingScore: number,
          communicationScore: number,
          objectionHandlingScore: number,
          closingScore: number,
          overallScore: number,
          strengths: [],
          weaknesses: [],
          improvementSuggestions: [],
          sentiment: "",
          clientInterestLevel: ""
        }

        Transcript:
        ${transcriptText}
        `;

        const analysisResponse = await openai.chat.completions.create({
            model: "gpt-4o-mini",
            messages: [{ role: "user", content: analysisPrompt }],
            temperature: 0.3
        });

        const analysisData = JSON.parse(analysisResponse.choices[0].message.content);

        // 3️⃣ Save to MongoDB
        const newCall = new Call({
            salespersonName,
            clientName,
            duration: "Auto detect later",
            transcript: transcriptText,
            analysis: analysisData,
            overallScore: analysisData.overallScore
        });

        await newCall.save();

        res.json({
            message: "Call analyzed successfully",
            data: newCall
        });

    } catch (error) {
        console.error(error);
        res.status(500).json({ error: "Something went wrong" });
    }
};

📄 call-upload.component.ts

import { HttpClient } from '@angular/common/http';

export class CallUploadComponent {

  salespersonName = '';
  clientName = '';
  selectedFile: any;
  report: any;

  constructor(private http: HttpClient) {}

  onFileChange(event: any) {
    this.selectedFile = event.target.files[0];
  }

  uploadCall() {
    const formData = new FormData();
    formData.append('audio', this.selectedFile);
    formData.append('salespersonName', this.salespersonName);
    formData.append('clientName', this.clientName);

    this.http.post('http://localhost:5000/api/calls/upload', formData)
      .subscribe((res: any) => {
        this.report = res.data;
      });
  }
}
